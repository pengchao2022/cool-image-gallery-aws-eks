#常用的执行命令：

-- 列出所有数据库
\l

-- 切换到另一个数据库
\c 数据库名

-- 列出当前数据库的所有表
\dt

-- 列出所有表（包括系统表）
\dt+

-- 查看表结构
\d 表名

-- 查看所有用户/角色
\du

-- 执行 SQL 文件
\i /path/to/file.sql

-- 显示查询执行时间
\timing

-- 退出 psql
\q
